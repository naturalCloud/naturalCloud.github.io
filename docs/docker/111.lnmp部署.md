编辑一个部署php的yaml，由下边软件构成
 * laravel8
 * nginx 
 * redis 
 * mysql8
 * php8


## dockerfile

### nginx

#### dockerfile
```dockerfile
FROM nginx:1.15

COPY ./app.conf /etc/nginx/conf.d/app.conf

```
#### conf

```nginx configuration
server {
    listen 80;
    listen [::]:80;
    server_name example.com;
    root /var/www/laravel/public;

    add_header X-Frame-Options "SAMEORIGIN";
    add_header X-Content-Type-Options "nosniff";

    index index.php;

    charset utf-8;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location = /favicon.ico { access_log off; log_not_found off; }
    location = /robots.txt  { access_log off; log_not_found off; }

    error_page 404 /index.php;

    location ~ \.php$ {
        fastcgi_pass app:9000;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
    }

    location ~ /\.(?!well-known).* {
        deny all;
    }
}

```
### php

```dockerfile
FROM php:8.0-fpm-buster

RUN pecl install redis \
    && docker-php-ext-enable redis


RUN curl -sS https://getcomposer.org/installer | php -- --install-dir=/usr/local/bin --filename=composer \
        && composer config -g repo.packagist composer https://packagist.phpcomposer.com \

```

### mysq8

FROM mysql:latest

```

### redis

FROM redis:6.0.8

```
## 完整 yaml

```yaml
version: '3.0'
services:
  app:
    build:
      context: ./
      dockerfile: php.Dockerfile
    container_name: app-fpm
    volumes:
      - ./:/var/www/laravel
    depends_on:
      - app-mysql
      - app-redis
    environment:
      - DB_HOST=app-mysql
      - REDIS_HOST=app-redis

  app-nginx:
    build:
      context: ./
      dockerfile: nginx.Dockerfile
    container_name: app-nginx
    volumes:
      - ./:/var/www/laravel
    ports:
      - "8082:80"
    depends_on:
      - app

  app-redis:
    build:
      context: ./
      dockerfile: redis.Dockerfile
    container_name: app-redis

  app-mysql:
    build:
      context: ./
      dockerfile: mysql.Dockerfile
    container_name: app-mysql
    environment:
      - MYSQL_ROOT_PASSWORD=123456
    volumes:
      - /home/zhangshuai/.mysql8/data:/var/lib/mysql
      - /home/zhangshuai/.mysql8/conf:/etc/mysql
      - /home/zhangshuai/.mysql8/logs:/var/log/mysql
      - /home/zhangshuai/.mysql8/mysql-files:/var/lib/mysql-files


```
